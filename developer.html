1. Authentication

Include your secret key in the X-API-Key header for every request.

curl -X GET "https://data4ghana.com/api/buy-data" \
  -H "X-API-Key: YOUR_SECRET_KEY"

2. API Endpoints

POST /api/buy-data

Purchase data for MTN, Telecel, or Ishare.

{

  "network": "MTN",

4. Order Status

Track the fulfillment status of any transaction.

GET /api/status/{orderId}
// Response: { "status": "processed", "id": "uuid" }

5. Wallet Balance

Check your current agent credit balance.

GET /api/balance/{agent_id}
// Response: { "balance": 250.50 }

6. Implementation Examples

Use these production-ready snippets to integrate order tracking into your application. Replace {orderId} with your actual transaction ID.

PHP (cURL) 
<?php
$apiKey = "sk_live_your_secret_key";
$orderId = "7f8a..."; // The ID returned from /buy-data

$ch = curl_init();
curl_setopt($ch, CURLOPT_URL, "https://data4ghana.com/api/v1/order/status/" . $orderId);
curl_setopt($ch, CURLOPT_RETURNTRANSFER, true);
curl_setopt($ch, CURLOPT_HTTPHEADER, [
    "X-API-Key: " . $apiKey,
    "Content-Type: application/json"
]);

$response = curl_exec($ch);
curl_close($ch);

$result = json_json_decode($response, true);
if ($result['status'] === 'success') {
    echo "Order Status: " . $result['data']['status'];
}
?>
Node.js (Axios) 
const axios = require('axios');

const checkStatus = async (orderId) => {
  try {
    const response = await axios.get(`https://data4ghana.com/api/v1/order/status/${orderId}`, {
      headers: {
        'X-API-Key': 'sk_live_your_secret_key'
      }
    });

    console.log('Order Details:', response.data.data);
  } catch (error) {
    console.error('Check failed:', error.response?.data || error.message);
  }
};

checkStatus('7f8a...');
